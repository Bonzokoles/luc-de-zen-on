  <button onclick="sendPrompt()">Wyślij</button>
  <div id="responseArea"></div>
</div>

<script type="module" is:inline>
  async function sendPrompt() {
    const prompt = document.getElementById('prompt').value.trim();
    const responseArea = document.getElementById('responseArea');
    responseArea.textContent = 'Ładowanie...';

    if (!prompt) {
      responseArea.textContent = 'Proszę wpisać zapytanie.';
      return;
    }

    try {
      const res = await fetch('/api/bielik', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt }),
      });
      const data = await res.json();

      if (data.error) {
        responseArea.innerHTML = `<span class="error">Błąd: ${data.error}</span>`;
      } else {
        responseArea.textContent = data.result;
      }
    } catch (e) {
      responseArea.textContent = 'Błąd połączenia: ' + e.message;
    }
  }
</script>
